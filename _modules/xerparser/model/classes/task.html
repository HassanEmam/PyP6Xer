    def duration(self):
        dur = None
        if self.target_drtn_hr_cnt:
            if self.calendar.day_hr_cnt:
                dur = self.target_drtn_hr_cnt / self.calendar.day_hr_cnt
            else:
                dur = self.target_drtn_hr_cnt / 8.0
        else:
            dur =0.0
        return dur
