    def find_by_id(self, id) -> Project:
        obj = list(filter(lambda x: x.proj_id == id, self._projects))
        if obj:
            return obj[0]
        return obj

    def __next__(self) -> Project:
        if self.index >= len(self._projects):
            raise StopIteration
        idx = self.index
        self.index +=1
        return self._projects[idx]
